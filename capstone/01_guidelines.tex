\chapter{Capstone Project Design Guidelines}
\emph{Business Math \& Finance: Applied Accounting with Excel}

\section{Vision \& Global Goals}

\begin{longtable}{|p{3cm}|p{5cm}|p{6cm}|}
\hline
\textbf{Anchor} & \textbf{Guiding Goal} & \textbf{Success Indicators} \\
\hline
\textbf{Authentic Integration} & Extend Semester-1 mini-projects into one investor-ready business plan and linked Excel workbook. & Three or more prior models imported; formulas traceable end-to-end; no hard-coded numbers in statements. \\
\hline
\textbf{Advanced Spreadsheet Literacy} & Demonstrate automation, validation, and scenario analysis commensurate with entry-level analyst skills. & Self-auditing dashboard; macro or button to refresh statements; Scenario Manager dashboard. \\
\hline
\textbf{Analytic Decision-Making} & Use data and sensitivity results to justify strategic choices. & Tornado chart identifies top three cash-flow drivers; narrative explains pivots. \\
\hline
\textbf{Professional Communication} & Deliver a concise, data-driven 10-slide pitch and a 3-min model-tour video. & Slides follow Problem-Solution-Market-Financials arc; model-tour clearly shows linkages. \\
\hline
\textbf{Reflection \& CAP Growth} & Cultivate Courage, Adaptability, Persistence through weekly journaling and peer critique. & 150-word reflections logged weekly; peer-review checklist completed thoughtfully. \\
\hline
\end{longtable}

\section{Component Objectives \& Depth Expectations}

\begin{longtable}{|p{3cm}|p{6cm}|p{5cm}|}
\hline
\textbf{Component} & \textbf{Core Objectives} & \textbf{Depth of Evidence Required} \\
\hline
\textbf{Concept Proposal (Week 1)} & \begin{itemize}
\item Frame a community-rooted problem ’ solution
\item Allocate team roles tied to expertise
\end{itemize} & 1-page proposal; preliminary Gantt auto-generated in Excel \\
\hline
\textbf{Market Research (Week 2)} & \begin{itemize}
\item Collect primary/secondary data
\item Apply descriptive stats to competitor pricing
\end{itemize} & 250-word research memo + Excel sheet with mean, median, Ã and conditional-formatting outliers \\
\hline
\textbf{Revenue Streams (Week 3)} & \begin{itemize}
\item Build unit-economics table
\item Deploy Goal Seek to hit target margin
\end{itemize} & Dynamic table pulling assumptions; commentary on break-even units \\
\hline
\textbf{Start-up Budget (Week 4)} & \begin{itemize}
\item Categorise CapEx vs. OpEx with SUMIF
\item Embed drop-down expense codes
\end{itemize} & Expense log rolls into monthly budget; error flag if category missing \\
\hline
\textbf{Funding Strategy (Week 5)} & \begin{itemize}
\item Compare equity vs. debt scenarios
\item Construct amortisation schedule with PMT/IPMT
\end{itemize} & 12-row equity cap-table + loan sheet feeding interest-expense line \\
\hline
\textbf{Pricing \& Forecast (Week 6)} & \begin{itemize}
\item Optimise price-volume using 2-variable data table
\item Graph CVP
\end{itemize} & Interactive slider or data-table; CVP chart auto-updates \\
\hline
\textbf{Payroll Plan (Week 7)} & \begin{itemize}
\item Design wage bands with XLOOKUP
\item Calculate gross’net with tax tables
\end{itemize} & Payroll register for at least two employee types; weekly and monthly summaries \\
\hline
\textbf{Inventory Planning (Week 8)} & \begin{itemize}
\item Code FIFO/LIFO macro
\item Compute turnover KPIs
\end{itemize} & Macro processes e 3 purchase \& sales lots; dashboard shows turnover \& days on hand \\
\hline
\textbf{Depreciation \& Asset Mgmt (Week 9)} & \begin{itemize}
\item Compare SLN vs. DDB schedules
\item Link to Balance Sheet
\end{itemize} & Dual-method schedule; toggle feeds financials \\
\hline
\textbf{Integrated Model (Week 10)} & \begin{itemize}
\item Assemble linked 12-month I/S, B/S, C/F
\item Build validation checks
\end{itemize} & All statements reconcile; traffic-light indicators for errors \\
\hline
\textbf{Sensitivity \& Risk (Week 11)} & \begin{itemize}
\item Run Scenario Manager (best/base/worst)
\item Generate tornado chart
\end{itemize} & Scenario summary sheet; chart ranks top 5 sensitivities \\
\hline
\textbf{Pitch Deck (Week 12)} & \begin{itemize}
\item Translate model insights into investor narrative
\end{itemize} & 10 slides; model screenshots annotated \\
\hline
\textbf{Demo Day \& Reflection (Week 13)} & \begin{itemize}
\item Deliver live pitch + model-tour video
\item Submit 500-word final reflection on CAP growth
\end{itemize} & Video d 3 min; written reflection cites concrete challenges overcome \\
\hline
\end{longtable}

\section{Summative Rubric Details}

\begin{longtable}{|p{3cm}|p{4cm}|p{4cm}|p{4cm}|}
\hline
\textbf{Domain} & \textbf{Exceeds (A)} & \textbf{Meets (B-C)} & \textbf{Emerging (D)} \\
\hline
\textbf{Model Fidelity \& Automation} (10) & All sheets linked; no hard-codes; macro refresh; validation dashboard clean & Statements linked; d 2 manual plugs; minor validation warnings & Broken links, manual totals, or major errors \\
\hline
\textbf{Analytic Insight \& Decision Quality} (10) & Sensitivity analysis triggers strategic pivots explained with data & Sensitivity analysis present; limited strategic discussion & Calculations present but not interpreted \\
\hline
\textbf{Documentation \& Sourcing} (5) & MLA citations; cell comments on all key formulas; AI use logged & Basic citation list; some formula notes & Missing citations or opaque formulas \\
\hline
\textbf{Investor-Ready Pitch} (10) & Storyline compelling; visuals reinforce numbers; within 9-11 min & Clear flow; minor design or timing issues & Disjointed or over-/under-time \\
\hline
\textbf{Peer-Critique Quality} (2) & Identifies root-cause errors; offers fixes & Flags surface issues & Minimal or perfunctory feedback \\
\hline
\end{longtable}

\section{Implementation Notes for Teachers}

\begin{itemize}
\item \textbf{Template Package}  Provide:
  \begin{itemize}
  \item "Capstone Workbook Shell" with pre-built tabs/headings
  \item Peer-review checklist (Google Form or Excel)
  \item Slide-deck master with section placeholders
  \end{itemize}

\item \textbf{Skill Refreshers}  20-min micro-lessons on Scenario Manager (Week 5) and macros (Week 8).

\item \textbf{Grading Efficiency}  Rely on validation dashboard signals and peer-review accuracy; spot-grade 30\% of formulas rather than all.
\end{itemize}
